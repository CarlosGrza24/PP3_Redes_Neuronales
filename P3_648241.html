from tensorflow.keras.layers import GlobalAveragePooling2D

def build_model_6():
    input_shape = (IMG_HEIGHT, IMG_WIDTH, 1)
    num_classes = train_generator.num_classes

    model = Sequential([
        # Bloque 1
